def create_frequency_df(src_df):
  temp_df = pd.DataFrame()
  temp_df['Rok'] = src_df.Rok.unique()
  temp_df.set_index('Rok', inplace=True)
  p_k = src_df[src_df['Zmienna'] == 'kobiety'].set_index('Rok')['Wartosc'] / 100
  p_o = src_df[src_df['Zmienna'] == 'ogem'].set_index('Rok')['Wartosc'] / 100
  l_k = df_gminy[(df_gminy['Kategoria'] == 'Kobiety wg grup wieku') & (df_gminy['Zmienna'] == 'ogem') & (df_gminy['Kod'] == src_df['Kod'].unique()[0])].set_index('Rok')['Wartosc']
  l_m = df_gminy[(df_gminy['Kategoria'] == 'Mczyni wg grup wieku') & (df_gminy['Zmienna'] == 'ogem') & (df_gminy['Kod'] == src_df['Kod'].unique()[0])].set_index('Rok')['Wartosc']
  temp_df['udzia kobiet w bezrobociu'] = p_k * l_k / (p_o * (l_k + l_m))
  temp_df['udzia mczyzn w bezrobociu'] = 1 - temp_df['udzia kobiet w bezrobociu']

  return temp_df

stds_list = list()
groups = df_bezrobocie.groupby(['Kod'])

for name, group in groups: 
  p_k = group[group['Zmienna'] == 'kobiety'].set_index('Rok')['Wartosc']
  p_m = group[group['Zmienna'] == 'mczyni'].set_index('Rok')['Wartosc']
  std = (p_k - p_m).std()
  temp_serie = {'Kod': name, 'std__udzia kobiet w bezrobociu': std}
  stds_list.append(temp_serie)

stds_list.sort(key=lambda x: x['std__udzia kobiet w bezrobociu'], reverse=True)

for i in range(3):
  code = stds_list[i]['Kod']
  group = groups.get_group(code)
